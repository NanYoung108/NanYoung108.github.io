if嵌套结构语句

#include<stdio.h>
iint main()
{

/*if嵌套结构语句（就是说满足条件时的if的执行代码里面还可以也可以再次使用if的条件语句）
目的是能够更精确的判断所描述的内容：
if（条件1）{若满足条件1时执行的代码}else if（条件2）{条件2满足时执行的代码}...else{}都不满足时执行的代码
题目：输入您的高考分数，若大于600分则视为考上一本，500到600之间视为考上二本，400到500之间视为考上三本，
400分以下视为没有考上本科,其中在一本分数中，如果大于700分则考入清华大学，若大于650则考入武汉大学*/
int score;
//1、提示输入高考分数
printf("您的高考分数为：");
//2、显示高考的分数(用输入设备（即键盘）输入高考分数)
scanf("%d",&score);
//3、开始判断 
if(score>600)
{printf("恭喜你考上了一本！");
if(score>700){printf("恭喜你能考上清华大学");}
else
if(score>650)
{printf("恭喜您能考上武汉大学"); 
} 
}
else
if(score>500)
{printf("恭喜你考上了二本！"); 
}else
if(score>400)
{printf("恭喜你考上了三本");
}else
{printf("很遗憾！你没有考上本科，请继续加油吧"); 
} 

return 0;

}
